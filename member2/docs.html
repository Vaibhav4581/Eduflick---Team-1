Lesson Sequential Unlock

isLessonUnlocked(lesson, index, module):
  if module.state === 'locked'
    return FALSE

  if index === 0
    return TRUE  // first lesson always open

  prev = module.lessons[index - 1]
  return lessonProgress[prev.id] === true

Assessment Unlock

assessmentAvailable(module):
  lessons = module.lessons
  return lessons.length > 0
    && lessons.every(l =>
         lessonProgress[l.id] === true
       )

Module Progression

getModuleState(module, index):
  if index === 0
    → computeModuleState(module)

  prev = modules[index - 1]
  prevResults = assessmentResults[
    prev.assessment.id
  ]
  prevPassed = prevResults.some(r =>
    r.score >= prev.passing_score
  )

  if NOT prevPassed → "locked"
  else → computeModuleState(module)
